    <script>
      var coins = 20;
      var slots = ['1', '2', '3', '4', '5'];
      var payouts = {
        '111': 2,
        '222': 3,
        '333': 4,
        '444': 5,
        '555': 10
      };
      
      function spin() {
        var bet = parseInt(document.getElementById('bet').value);
        if (bet > coins) {
          alert('You do not have enough coins to place that bet.');
          return;
        }
        coins -= bet;
        document.getElementById('coins').innerHTML = coins;
        var slot1 = getRandomSlot();
        var slot2 = getRandomSlot();
        var slot3 = getRandomSlot();
        document.getElementById('slot1').innerHTML = slot1;
        document.getElementById('slot2').innerHTML = slot2;
        document.getElementById('slot3').innerHTML = slot3;
        var result = slot1 + slot2 + slot3;
        document.getElementById('result').innerHTML = result;
        if (payouts[result]) {
          var payout = bet * payouts[result];
          coins += payout;
          document.getElementById('coins').innerHTML = coins;
          alert('You won ' + payout + ' coins!');
        }
      }
      
      function getRandomSlot() {
        return slots[Math.floor(Math.random() * slots.length)];
      }
    </script>
